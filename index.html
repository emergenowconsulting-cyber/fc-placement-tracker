// ============================================================ // Fruitful Communities — Placement Tracker // Google Apps Script Backend v2 // Fixes: CORS handling, POST via GET workaround // ============================================================ const SHEET_NAME = 'Sheet1'; const REPORT_EMAIL = 'Deborah@fruitfulcommunities.ca'; // ── CORS helper ────────────────────────────────────────── function corsOutput(data) { const output = ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); return output; } // ── ALL requests come through doGet ────────────────────── // We use GET for everything to avoid CORS preflight issues // Parameters are passed as URL query params function doGet(e) { try { const p = e.parameter; const action = p.action; if (action === 'ping') { return corsOutput({ success: true, message: 'Connected', time: Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'yyyy-MM-dd HH:mm:ss') }); } if (action === 'getShifts') { return getShifts(); } if (action === 'submitShift') { return submitShift(p); } if (action === 'saveProgress') { return saveProgress(p); } return corsOutput({ success: false, error: 'Unknown action: ' + action }); } catch (err) { return corsOutput({ success: false, error: err.message, stack: err.stack }); } } // Keep doPost as fallback function doPost(e) { return doGet(e); } // ── GET all shifts ─────────────────────────────────────── function getShifts() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) return corsOutput({ success: false, error: 'Sheet not found: ' + SHEET_NAME }); // Ensure headers exist ensureHeaders(sheet); const data = sheet.getDataRange().getValues(); if (data.length <= 1) return corsOutput({ success: true, shifts: [] }); const headers = data[0]; const shifts = data.slice(1) .filter(row => row[0]) // skip empty rows .map(row => { const obj = {}; headers.forEach((h, i) => { const key = headerToKey(h.toString()); obj[key] = row[i] !== undefined ? row[i].toString() : ''; }); return obj; }); return corsOutput({ success: true, shifts: shifts }); } // ── Convert header to camelCase key ───────────────────── function headerToKey(h) { return h .trim() .replace(/[^a-zA-Z0-9\s]/g, '') .replace(/\s+(.)/g, function(_, c) { return c.toUpperCase(); }) .replace(/^(.)/, function(c) { return c.toLowerCase(); }); } // ── SUBMIT a completed shift ────────────────────────────── function submitShift(p) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) return corsOutput({ success: false, error: 'Sheet not found' }); const shiftId = p.shiftId || (p.firstName + '_' + p.lastName + '_' + Date.now()).toLowerCase().replace(/\s/g,''); const existingRow = findRowByShiftId(sheet, shiftId); const row = [ shiftId, p.date || '', p.firstName || '', p.lastName || '', p.program || '', p.clockIn || '', p.clockOut || '', p.hours || '0', 'Submitted', p.areas || '', p.techSkills || '', p.softSkills || '', p.skillsHow || '', p.frameworks || '', p.frameworkHow || '', p.summary || '', p.challenges || '', p.clients || '0', Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'yyyy-MM-dd HH:mm:ss') ]; if (existingRow > 0) { const range = sheet.getRange(existingRow, 1, 1, row.length); range.setNumberFormat('@STRING@'); // store as plain text, prevents date serial conversion range.setValues([row]); } else { sheet.appendRow(row); // Format the new row as plain text const newRow = sheet.getLastRow(); sheet.getRange(newRow, 1, 1, row.length).setNumberFormat('@STRING@'); } // Try to send email notification try { sendDailyReport('today'); } catch(emailErr) { Logger.log('Email error: ' + emailErr.message); } return corsOutput({ success: true, message: 'Shift submitted', shiftId: shiftId }); } // ── SAVE progress (draft) ──────────────────────────────── function saveProgress(p) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) return corsOutput({ success: false, error: 'Sheet not found' }); const shiftId = p.shiftId || (p.firstName + '_' + p.lastName + '_' + Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'yyyy-MM-dd HH:mm:ss').split('T')[0]).toLowerCase().replace(/\s/g,''); const existingRow = findRowByShiftId(sheet, shiftId); const row = [ shiftId, p.date || '', p.firstName || '', p.lastName || '', p.program || '', p.clockIn || '', p.clockOut || '', '', 'In Progress', p.areas || '', p.techSkills || '', p.softSkills || '', p.skillsHow || '', p.frameworks || '', p.frameworkHow || '', p.summary || '', p.challenges || '', p.clients || '0', Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'yyyy-MM-dd HH:mm:ss') ]; if (existingRow > 0) { const range2 = sheet.getRange(existingRow, 1, 1, row.length); range2.setNumberFormat('@STRING@'); range2.setValues([row]); } else { sheet.appendRow(row); const newRow2 = sheet.getLastRow(); sheet.getRange(newRow2, 1, 1, row.length).setNumberFormat('@STRING@'); } return corsOutput({ success: true, message: 'Progress saved', shiftId: shiftId }); } // ── Find a row by Shift ID ─────────────────────────────── function ensureHeaders(sheet) { const first = sheet.getRange(1,1).getValue(); if (!first || first.toString().trim() === '') { const headers = ['Shift ID','Date','First Name','Last Name','Program', 'Clock In','Clock Out','Hours','Status','Areas','Technical Skills', 'Soft Skills','Skills How','Frameworks','Framework How','Summary', 'Challenges','Clients','Submitted At']; const headerRange = sheet.getRange(1, 1, 1, headers.length); headerRange.setValues([headers]); headerRange.setFontWeight('bold'); headerRange.setBackground('#1B5AAB'); headerRange.setFontColor('#ffffff'); } } function findRowByShiftId(sheet, shiftId) { if (!shiftId) return -1; const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0].toString() === shiftId.toString()) return i + 1; } return -1; } // ── DAILY EMAIL REPORT ─────────────────────────────────── function sendDailyReport(period) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); const data = sheet.getDataRange().getValues(); if (data.length <= 1) return; const today = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'yyyy-MM-dd HH:mm:ss').split('T')[0]; const weekAgo = new Date(Date.now() - 7*86400000).toISOString().split('T')[0]; const headers = data[0]; const allRows = data.slice(1).map(row => { const obj = {}; headers.forEach((h, i) => { obj[h] = row[i] ? row[i].toString() : ''; }); return obj; }); let subset = allRows.filter(r => r['Status'] === 'Submitted'); if (period === 'today') subset = subset.filter(r => r['Date'] === today); if (period === 'week') subset = subset.filter(r => r['Date'] >= weekAgo); if (!subset.length) return; const totalHours = subset.reduce((a,r) => a + (parseFloat(r['Hours'])||0), 0); const totalClients = subset.reduce((a,r) => a + (parseInt(r['Clients'])||0), 0); const students = [...new Set(subset.map(r => r['First Name'] + ' ' + r['Last Name']))]; const areas = [...new Set(subset.flatMap(r => (r['Areas']||'').split(',').map(a=>a.trim()).filter(Boolean)))]; const rows = subset.map(r => ` ${r['First Name']} ${r['Last Name']} ${r['Program']||''} ${r['Areas']||'-'} ${parseFloat(r['Hours']||0).toFixed(1)}h ${r['Clients']||0} `).join(''); const html = `
Fruitful Communities
Placement Report - ${today}

Dear Deborah,

Here is today's placement activity summary.

${subset.length}
Shifts
${totalHours.toFixed(1)}h
Hours
${totalClients}
Clients
${students.length}
Students
Areas: ${areas.join(', ')||'-'}
Students: ${students.join(', ')||'-'}
${rows}
Student Program Area Hours Clients

Warm regards,
Fruitful Communities Placement Tracking System

Auto-generated ${new Date().toLocaleString()}

`; GmailApp.sendEmail( REPORT_EMAIL, 'Fruitful Communities - Daily Placement Report - ' + today, 'Please view this email in HTML format.', { htmlBody: html } ); } // ── SCHEDULED TRIGGER ──────────────────────────────────── function sendScheduledReport() { sendDailyReport('today'); }
